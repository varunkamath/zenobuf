1//! Message trait and utilities for working with Protocol Buffer messages
2
3// No imports needed
4use prost::Message as ProstMessage;
5
6use crate::error::{Error, Result};
7
8/// Trait for Zenobuf messages
9///
10/// This trait is implemented for all Protocol Buffer messages that can be used
11/// with the Zenobuf framework. It provides methods for serialization and
12/// deserialization, as well as type information.
13///
14/// Users can implement this trait for their own Protocol Buffer messages, or
15/// use the `ZenobufMessage` derive macro from the `zenobuf-macros` crate.
16pub trait Message: ProstMessage + Default + Clone + Send + Sync + 'static {
17    /// Returns the type name of the message
18    ///
19    /// This is used for type checking and debugging.
20    fn type_name() -> &'static str;
21
22    /// Decodes a message from a byte slice
23    ///
24    /// This is a convenience method that calls `prost::Message::decode`.
25    fn decode_from_slice(bytes: &[u8]) -> Result<Self> {
26        Self::decode(bytes).map_err(Error::from)
27    }
28}
29
30/// Helper function to encode a message to a byte vector
31pub fn encode_message<M: Message>(message: &M) -> Vec<u8> {
32    let mut buf = Vec::with_capacity(message.encoded_len());
33    message.encode(&mut buf).expect("Failed to encode message");
34    buf
35}
36
37/// Helper function to decode a message from a byte slice
38pub fn decode_message<M: Message>(bytes: &[u8]) -> Result<M> {
39    M::decode_from_slice(bytes)
40}
41
42/// Helper function to get the type name of a message
43pub fn message_type_name<M: Message>() -> &'static str {
44    M::type_name()
45}
