13/// A boxed future for async operations
14pub type BoxFuture<'a, T> = Pin<Box<dyn Future<Output = T> + Send + 'a>>;
15
16/// Transport layer abstraction
17///
18/// This trait defines the interface that all transport implementations must provide.
19/// It allows for pluggable transport layers while maintaining a consistent API.
20#[async_trait::async_trait]
21pub trait Transport: Send + Sync + 'static {
22    /// Create a publisher for the given topic
23    async fn create_publisher<M: Message>(
24        &self,
25        topic: &str,
26    ) -> Result<Arc<crate::publisher::Publisher<M>>>;
27
28    /// Create a subscriber for the given topic with a callback
29    async fn create_subscriber<M: Message, F>(
30        &self,
31        topic: &str,
32        callback: F,
33    ) -> Result<Arc<crate::subscriber::Subscriber>>
34    where
35        F: Fn(M) + Send + Sync + 'static;
36
37    /// Create a service for the given service name with a handler
38    async fn create_service<Req: Message, Res: Message, F>(
39        &self,
40        service_name: &str,
41        handler: F,
42    ) -> Result<Arc<crate::service::Service>>
43    where
44        F: Fn(Req) -> Result<Res> + Send + Sync + 'static;
45
46    /// Create a client for the given service name
47    fn create_client<Req: Message, Res: Message>(
48        &self,
49        service_name: &str,
50    ) -> Result<Arc<crate::client::Client<Req, Res>>>;
51}
52
53/// Publisher abstraction
54pub trait Publisher<M: Message>: Send + Sync + 'static {
55    /// Publishes a message
56    fn publish(&self, message: &M) -> Result<()>;
57}
58
59/// Subscriber abstraction
60pub trait Subscriber: Send + Sync + 'static {
61    /// Closes the subscriber
62    fn close(&self) -> Result<()>;
63}
64
65/// Service abstraction
66pub trait Service: Send + Sync + 'static {
67    /// Closes the service
68    fn close(&self) -> Result<()>;
69}
70
71/// Client abstraction
72pub trait Client<Req: Message, Res: Message>: Send + Sync + 'static {
73    /// Calls the service with the given request
74    fn call(&self, request: &Req) -> Result<Res>;
75
76    /// Calls the service with the given request asynchronously
77    fn call_async<'a>(&'a self, request: &'a Req) -> BoxFuture<'a, Result<Res>>;
78}
