16/// Node abstraction for Zenobuf
17///
18/// A Node is the main entry point for using Zenobuf. It provides methods for
19/// creating publishers, subscribers, services, and clients.
20pub struct Node {
21    /// Name of the node
22    name: String,
23    /// Transport layer
24    transport: ZenohTransport,
25    /// Publishers
26    publishers: Mutex<HashMap<String, Box<dyn std::any::Any + Send + Sync>>>,
27    /// Subscribers
28    subscribers: Mutex<HashMap<String, Box<dyn std::any::Any + Send + Sync>>>,
29    /// Services
30    services: Mutex<HashMap<String, Box<dyn std::any::Any + Send + Sync>>>,
31    /// Clients
32    clients: Mutex<HashMap<String, Box<dyn std::any::Any + Send + Sync>>>,
33    /// Parameters
34    parameters: Mutex<HashMap<String, Parameter>>,
35}
36
37impl Node {
38    /// Creates a new Node with the given name
39    pub async fn new(name: &str) -> Result<Self> {
40        let transport = ZenohTransport::new().await?;
41        Self::with_transport(name, transport)
42    }
43
44    /// Creates a new Node with the given name and transport
45    pub fn with_transport(name: &str, transport: ZenohTransport) -> Result<Self> {
46        Ok(Self {
47            name: name.to_string(),
48            transport,
49            publishers: Mutex::new(HashMap::new()),
50            subscribers: Mutex::new(HashMap::new()),
51            services: Mutex::new(HashMap::new()),
52            clients: Mutex::new(HashMap::new()),
53            parameters: Mutex::new(HashMap::new()),
54        })
55    }

57    /// Returns the name of the node
58    pub fn name(&self) -> &str {

62    /// Creates a publisher for the given topic
63    pub async fn create_publisher<M: Message>(
64        &self,
65        topic: &str,
66        _qos: QosProfile,
67    ) -> Result<Arc<Publisher<M>>> {
68        // Use the topic name as provided by the user (global topics by default)
69        let topic_name = topic.to_string();
70
71        // Check if the publisher already exists
72        {
73            let publishers = self.publishers.lock().unwrap();
74            if publishers.contains_key(&topic_name) {
75                return Err(Error::TopicAlreadyExists(topic_name));
76            }
77        } // MutexGuard is dropped here
78
79        // Create the publisher
80        let inner_publisher = self.transport.create_publisher::<M>(&topic_name).await?;
81        let publisher = Arc::new(Publisher::new(
82            topic_name.clone(),
83            Box::new(inner_publisher),
84        ));
85
86        // Store the publisher
87        let mut publishers = self.publishers.lock().unwrap();
88        publishers.insert(topic_name, Box::new(publisher.clone()));
89
90        Ok(publisher)
91    }

93    /// Creates a subscriber for the given topic with a callback
94    pub async fn create_subscriber<M: Message, F>(
95        &self,
96        topic: &str,
97        _qos: QosProfile,
98        callback: F,
99    ) -> Result<Arc<Subscriber>>
100    where
101        F: Fn(M) + Send + Sync + 'static,
102    {
103        // Use the topic name as provided by the user (global topics by default)
104        let topic_name = topic.to_string();
105
106        // Check if the subscriber already exists
107        {
108            let subscribers = self.subscribers.lock().unwrap();
109            if subscribers.contains_key(&topic_name) {
110                return Err(Error::TopicAlreadyExists(topic_name));
111            }
112        } // MutexGuard is dropped here
113
114        // Create the subscriber
115        let inner_subscriber = self
116            .transport
117            .create_subscriber::<M, F>(&topic_name, callback)
118            .await?;
119        let subscriber = Arc::new(Subscriber::new(
120            topic_name.clone(),
121            Box::new(inner_subscriber),
122        ));
123
124        // Store the subscriber
125        let mut subscribers = self.subscribers.lock().unwrap();
126        subscribers.insert(topic_name, Box::new(subscriber.clone()));
127
128        Ok(subscriber)
129    }

131    /// Creates a service for the given name with a handler
132    pub async fn create_service<Req: Message, Res: Message, F>(
133        &self,
134        service_name: &str,
135        handler: F,
136    ) -> Result<Arc<Service>>
137    where
138        F: Fn(Req) -> Result<Res> + Send + Sync + 'static,
139    {
140        // Use the service name as provided by the user (global services by default)
141        let full_service_name = service_name.to_string();
142
143        // Check if the service already exists
144        {
145            let services = self.services.lock().unwrap();
146            if services.contains_key(&full_service_name) {
147                return Err(Error::ServiceAlreadyExists(full_service_name));
148            }
149        } // MutexGuard is dropped here
150
151        // Create the service
152        let inner_service = self
153            .transport
154            .create_service::<Req, Res, F>(&full_service_name, handler)
155            .await?;
156        let service = Arc::new(Service::new(
157            full_service_name.clone(),
158            Box::new(inner_service),
159        ));
160
161        // Store the service
162        let mut services = self.services.lock().unwrap();
163        services.insert(full_service_name, Box::new(service.clone()));
164
165        Ok(service)
166    }

168    /// Creates a client for the given service name
169    pub fn create_client<Req: Message, Res: Message>(
170        &self,
171        service_name: &str,
172    ) -> Result<Arc<Client<Req, Res>>> {
173        // Use the service name as provided by the user (global services by default)
174        let full_service_name = service_name.to_string();
175
176        // Check if the client already exists
177        let mut clients = self.clients.lock().unwrap();
178        if clients.contains_key(&full_service_name) {
179            return Err(Error::ServiceAlreadyExists(full_service_name));
180        }
181
182        // Create the client
183        let inner_client = self
184            .transport
185            .create_client::<Req, Res>(&full_service_name)?;
186        let client = Arc::new(Client::new(
187            full_service_name.clone(),
188            Box::new(inner_client),
189        ));
190
191        // Store the client
192        clients.insert(full_service_name, Box::new(client.clone()));
193
194        Ok(client)
195    }

197    /// Sets a parameter
198    pub fn set_parameter<
199        T: serde::Serialize + serde::de::DeserializeOwned + Clone + Send + Sync + 'static,
200    >(
201        &self,
202        name: &str,
203        value: T,
204    ) -> Result<()> {
205        let mut parameters = self.parameters.lock().unwrap();
206        parameters.insert(name.to_string(), Parameter::new(name, value)?);
207        Ok(())
208    }
209
210    /// Gets a parameter
211    pub fn get_parameter<T: serde::de::DeserializeOwned + Clone + Send + Sync + 'static>(
212        &self,
213        name: &str,
214    ) -> Result<T> {
215        let parameters = self.parameters.lock().unwrap();
216        if let Some(param) = parameters.get(name) {
217            param.get_value()
218        } else {
219            Err(Error::Parameter(format!("Parameter not found: {}", name)))
220        }
221    }
