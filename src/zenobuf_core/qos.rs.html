1//! Quality of Service (QoS) profiles for Zenobuf
2
3use std::time::Duration;
4
5/// QoS preset for common use cases
6///
7/// This enum provides convenient presets for common QoS configurations,
8/// making it easier to configure quality of service without verbose setup.
9#[derive(Debug, Clone)]
10pub enum QosPreset {
11    /// Default QoS profile - reliable, volatile, keep last 10
12    Default,
13    /// Optimized for sensor data - best effort, volatile, keep last 5
14    SensorData,
15    /// Optimized for parameters - reliable, transient local, keep last 1
16    Parameters,
17    /// Optimized for services - reliable, volatile, keep last 10, 1s deadline
18    Services,
19    /// High throughput - best effort, volatile, keep last 100
20    HighThroughput,
21    /// Low latency - best effort, volatile, keep last 1
22    LowLatency,
23    /// Custom QoS profile
24    Custom(QosProfile),
25}
26
27/// Quality of Service profile for publishers and subscribers
28///
29/// This struct defines the quality of service parameters for publishers and
30/// subscribers. It is similar to the QoS profiles in ROS.
31#[derive(Debug, Clone)]
32pub struct QosProfile {
33    /// Reliability of the communication
34    pub reliability: Reliability,
35    /// Durability of the communication
36    pub durability: Durability,
37    /// History policy
38    pub history: History,
39    /// Depth of the history queue
40    pub depth: usize,
41    /// Deadline for receiving messages
42    pub deadline: Option<Duration>,
43    /// Lifespan of messages
44    pub lifespan: Option<Duration>,
45}
46
47impl Default for QosProfile {
48    fn default() -> Self {
49        Self {
50            reliability: Reliability::Reliable,
51            durability: Durability::Volatile,
52            history: History::KeepLast,
53            depth: 10,
54            deadline: None,
55            lifespan: None,
56        }
57    }
58}
59
60impl QosProfile {
61    /// Creates a new QoS profile with default values
62    pub fn new() -> Self {
63        Self::default()
64    }
65
66    /// Sets the reliability of the QoS profile
67    pub fn reliability(mut self, reliability: Reliability) -> Self {
68        self.reliability = reliability;
69        self
70    }
71
72    /// Sets the durability of the QoS profile
73    pub fn durability(mut self, durability: Durability) -> Self {
74        self.durability = durability;
75        self
76    }
77
78    /// Sets the history policy of the QoS profile
79    pub fn history(mut self, history: History) -> Self {
80        self.history = history;
81        self
82    }
83
84    /// Sets the depth of the history queue
85    pub fn depth(mut self, depth: usize) -> Self {
86        self.depth = depth;
87        self
88    }
89
90    /// Sets the deadline for receiving messages
91    pub fn deadline(mut self, deadline: Duration) -> Self {
92        self.deadline = Some(deadline);
93        self
94    }
95
96    /// Sets the lifespan of messages
97    pub fn lifespan(mut self, lifespan: Duration) -> Self {
98        self.lifespan = Some(lifespan);
99        self
100    }
101
102    /// Creates a QoS profile for sensors
103    ///
104    /// This profile is optimized for sensor data, which is typically
105    /// high-frequency and where the latest data is most important.
106    pub fn sensor_data() -> Self {
107        Self {
108            reliability: Reliability::BestEffort,
109            durability: Durability::Volatile,
110            history: History::KeepLast,
111            depth: 5,
112            deadline: None,
113            lifespan: None,
114        }
115    }
116
117    /// Creates a QoS profile for parameters
118    ///
119    /// This profile is optimized for parameters, which are typically
120    /// low-frequency and where reliability is important.
121    pub fn parameters() -> Self {
122        Self {
123            reliability: Reliability::Reliable,
124            durability: Durability::TransientLocal,
125            history: History::KeepLast,
126            depth: 1,
127            deadline: None,
128            lifespan: None,
129        }
130    }
131
132    /// Creates a QoS profile for services
133    ///
134    /// This profile is optimized for services, which require reliable
135    /// communication.
136    pub fn services() -> Self {
137        Self {
138            reliability: Reliability::Reliable,
139            durability: Durability::Volatile,
140            history: History::KeepLast,
141            depth: 10,
142            deadline: Some(Duration::from_secs(1)),
143            lifespan: None,
144        }
145    }
146}
147
148/// Reliability of the communication
149#[derive(Debug, Clone, Copy, PartialEq, Eq)]
150pub enum Reliability {
151    /// Best effort delivery (may drop messages)
152    BestEffort,
153    /// Reliable delivery (guaranteed delivery)
154    Reliable,
155}
156
157/// Durability of the communication
158#[derive(Debug, Clone, Copy, PartialEq, Eq)]
159pub enum Durability {
160    /// Volatile durability (no persistence)
161    Volatile,
162    /// Transient local durability (persistence on the publisher side)
163    TransientLocal,
164}
165
166/// History policy
167#[derive(Debug, Clone, Copy, PartialEq, Eq)]
168pub enum History {
169    /// Keep the last N messages
170    KeepLast,
171    /// Keep all messages
172    KeepAll,
173}
174
175impl From<QosPreset> for QosProfile {
176    fn from(preset: QosPreset) -> Self {
177        match preset {
178            QosPreset::Default => QosProfile::default(),
179            QosPreset::SensorData => QosProfile::sensor_data(),
180            QosPreset::Parameters => QosProfile::parameters(),
181            QosPreset::Services => QosProfile::services(),
182            QosPreset::HighThroughput => QosProfile {
183                reliability: Reliability::BestEffort,
184                durability: Durability::Volatile,
185                history: History::KeepLast,
186                depth: 100,
187                deadline: None,
188                lifespan: None,
189            },
190            QosPreset::LowLatency => QosProfile {
191                reliability: Reliability::BestEffort,
192                durability: Durability::Volatile,
193                history: History::KeepLast,
194                depth: 1,
195                deadline: None,
196                lifespan: None,
197            },
198            QosPreset::Custom(profile) => profile,
199        }
200    }
201}
202
203impl Default for QosPreset {
204    fn default() -> Self {
205        QosPreset::Default
206    }
