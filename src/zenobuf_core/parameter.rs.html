1//! Parameter system for Zenobuf
2
3use std::any::Any;
4use std::sync::{Arc, Mutex};
5
6use serde::{de::DeserializeOwned, Serialize};
7
8use crate::error::{Error, Result};
9
10/// Parameter for Zenobuf
11///
12/// A Parameter is a named value that can be set and retrieved.
13pub struct Parameter {
14    /// Name of the parameter
15    name: String,
16    /// Value of the parameter
17    value: Arc<Mutex<Box<dyn Any + Send + Sync>>>,
18    /// Serialized value of the parameter
19    serialized: Arc<Mutex<String>>,
20}
21
22impl Parameter {
23    /// Creates a new Parameter
24    pub fn new<T: Serialize + DeserializeOwned + Clone + Send + Sync + 'static>(
25        name: &str,
26        value: T,
27    ) -> Result<Self> {
28        let serialized = serde_json::to_string(&value)
29            .map_err(|e| Error::parameter(name, format!("Failed to serialize: {e}")))?;
30
31        Ok(Self {
32            name: name.to_string(),
33            value: Arc::new(Mutex::new(Box::new(value))),
34            serialized: Arc::new(Mutex::new(serialized)),
35        })
36    }
37
38    /// Returns the name of the parameter
39    pub fn name(&self) -> &str {
40        &self.name
41    }
42
43    /// Returns the value of the parameter
44    pub fn get_value<T: DeserializeOwned + Clone + Send + Sync + 'static>(&self) -> Result<T> {
45        let value = self.value.lock().unwrap();
46
47        // Try to downcast the value
48        if let Some(typed_value) = value.downcast_ref::<T>() {
49            return Ok(typed_value.clone());
50        }
51
52        // If downcasting fails, try to deserialize from the serialized value
53        let serialized = self.serialized.lock().unwrap();
54        let deserialized = serde_json::from_str::<T>(&serialized)
55            .map_err(|e| Error::parameter(&self.name, format!("Failed to deserialize: {e}")))?;
56
57        Ok(deserialized)
58    }
59
60    /// Sets the value of the parameter
61    pub fn set_value<T: Serialize + DeserializeOwned + Clone + Send + Sync + 'static>(
62        &self,
63        value: T,
64    ) -> Result<()> {
65        let serialized = serde_json::to_string(&value)
66            .map_err(|e| Error::parameter(&self.name, format!("Failed to serialize: {e}")))?;
67
68        *self.value.lock().unwrap() = Box::new(value);
69        *self.serialized.lock().unwrap() = serialized;
70
71        Ok(())
72    }
73}
